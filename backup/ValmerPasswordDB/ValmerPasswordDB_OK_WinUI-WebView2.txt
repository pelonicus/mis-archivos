    public sealed partial class MainWindow : Window

        private bool _yaInicieAutenticacion = false;
        private bool _accesoConcedido = false;
        private XDocument? _xmlData;
        private const string MasterKey = "V4lmerSys_Secur1ty_Key_2026_Auth!";
        private Dictionary<TreeViewNode, XElement> _nodosServidores = new Dictionary<TreeViewNode, XElement>();
        private Dictionary<TreeViewNode, string> _nodosGrupos = new Dictionary<TreeViewNode, string>();

        public MainWindow()
        {
            this.InitializeComponent();
            this.Title = "ValmerPasswordsDB";
            this.Activated += MainWindow_Activated;
        }

        // --- Gestión de Visibilidad de Contraseña ---
        private void BtnVerPassword_PointerPressed(object sender, PointerRoutedEventArgs e) => TxtPassword.PasswordRevealMode = PasswordRevealMode.Visible;
        private void BtnVerPassword_PointerReleased(object sender, PointerRoutedEventArgs e) => TxtPassword.PasswordRevealMode = PasswordRevealMode.Hidden;
        private void TxtPassword_PreventInput(object sender, KeyRoutedEventArgs e) => e.Handled = true;

        private async void MainWindow_Activated(object sender, WindowActivatedEventArgs args)
        {
            if (_yaInicieAutenticacion || _accesoConcedido) return;
            _yaInicieAutenticacion = true;
            await Task.Delay(500);
            if (await IniciarSeguridad())
            {
                _accesoConcedido = true;
                PanelBloqueo.Visibility = Visibility.Collapsed;
                PanelBienvenida.Visibility = Visibility.Visible;
                CargarDatosXML();
            }
            else { this.Close(); }
            _yaInicieAutenticacion = false;
        }

        private async Task<bool> IniciarSeguridad()
        {
            try
            {
                var disponibilidad = await global::Windows.Security.Credentials.UI.UserConsentVerifier.CheckAvailabilityAsync();
                if (disponibilidad == global::Windows.Security.Credentials.UI.UserConsentVerifierAvailability.Available)
                {
                    var resultado = await global::Windows.Security.Credentials.UI.UserConsentVerifier.RequestVerificationAsync("Acceso a ValmerPasswordsDB");
                    return resultado == global::Windows.Security.Credentials.UI.UserConsentVerificationResult.Verified;
                }
                return true;
            }
            catch { return false; }
        }

        private void CargarDatosXML()
        {
            try
            {
                string? onedrive = Environment.GetEnvironmentVariable("OneDriveCommercial") ?? Environment.GetEnvironmentVariable("OneDrive");
                if (string.IsNullOrEmpty(onedrive)) return;
                string ruta = Path.Combine(onedrive, "General - Servicios", "Softwares", "ValmerSystem", "ValmerPasswordsDB", "passwordsdb.xml");

                if (File.Exists(ruta))
                {
                    _xmlData = XDocument.Load(ruta);
                    ArbolServidores.RootNodes.Clear();
                    _nodosServidores.Clear();
                    _nodosGrupos.Clear();

                    foreach (var grupoXml in _xmlData.Descendants("Grupo"))
                    {
                        var nodoGrupo = new TreeViewNode() { Content = grupoXml.Attribute("nombre")?.Value ?? "Grupo" };
                        _nodosGrupos.Add(nodoGrupo, nodoGrupo.Content.ToString());
                        foreach (var srvXml in grupoXml.Descendants("Servidor"))
                        {
                            var nodoSrv = new TreeViewNode() { Content = $"{srvXml.Attribute("nombre")?.Value} ({srvXml.Attribute("ip")?.Value})" };
                            _nodosServidores.Add(nodoSrv, srvXml);
                            nodoGrupo.Children.Add(nodoSrv);
                        }
                        nodoGrupo.IsExpanded = true;
                        ArbolServidores.RootNodes.Add(nodoGrupo);
                    }
                }
            }
            catch (Exception ex) { System.Diagnostics.Debug.WriteLine(ex.Message); }
        }

        private void GuardarXML()
        {
            try
            {
                string? onedrive = Environment.GetEnvironmentVariable("OneDriveCommercial") ?? Environment.GetEnvironmentVariable("OneDrive");
                if (string.IsNullOrEmpty(onedrive) || _xmlData == null) return;
                string ruta = Path.Combine(onedrive, "General - Servicios", "Softwares", "ValmerSystem", "ValmerPasswordsDB", "passwordsdb.xml");
                _xmlData.Save(ruta);
            }
            catch { }
        }

        private void ArbolServidores_ItemInvoked(TreeView sender, TreeViewItemInvokedEventArgs args)
        {
            var nodo = args.InvokedItem as TreeViewNode;
            if (nodo != null && _nodosServidores.ContainsKey(nodo))
            {
                XElement srv = _nodosServidores[nodo];
                TxtTituloServidor.Text = $"{srv.Attribute("nombre")?.Value} ({srv.Attribute("ip")?.Value})";
                var usuario = srv.Descendants("Usuario").FirstOrDefault();
                if (usuario != null)
                {
                    TxtUsuario.Text = usuario.Attribute("nombre")?.Value;
                    var passElement = usuario.Elements("Password").FirstOrDefault(p => p.Attribute("activa")?.Value == "true");
                    TxtPassword.Password = Desencriptar(passElement?.Value ?? "");
                    BtnCambiarPass.IsEnabled = true;
                    BtnHistorial.IsEnabled = true;
                }
                TxtSSH.Text = srv.Descendants("Llave").FirstOrDefault(x => x.Attribute("activa")?.Value == "true")?.Element("Contenido")?.Value ?? "Sin llave SSH";
            }
        }

        private async void BtnCambiarPassword_Click(object sender, RoutedEventArgs e)
        {
            var nodo = ArbolServidores.SelectedNode;
            if (nodo == null || !_nodosServidores.ContainsKey(nodo)) return;
            XElement srv = _nodosServidores[nodo];
            XElement? usuario = srv.Descendants("Usuario").FirstOrDefault();
            if (usuario == null) return;

            // Construcción del diálogo con verificación
            var stack = new StackPanel { Spacing = 10, Width = 300 };
            var tPass1 = new PasswordBox { Header = "Nueva Contraseña", PasswordChar = "●" };
            var tPass2 = new PasswordBox { Header = "Confirmar Contraseña", PasswordChar = "●" };
            var tError = new TextBlock { Foreground = new Microsoft.UI.Xaml.Media.SolidColorBrush(Microsoft.UI.Colors.Red), Visibility = Visibility.Collapsed, FontSize = 12 };

            stack.Children.Add(tPass1);
            stack.Children.Add(tPass2);
            stack.Children.Add(tError);

            var dialog = new ContentDialog
            {
                Title = "Actualizar Seguridad",
                Content = stack,
                PrimaryButtonText = "Guardar",
                CloseButtonText = "Cancelar",
                XamlRoot = this.Content.XamlRoot
            };

            bool validado = false;
            while (!validado)
            {
                var result = await dialog.ShowAsync();
                if (result == ContentDialogResult.Primary)
                {
                    if (string.IsNullOrWhiteSpace(tPass1.Password))
                    {
                        tError.Text = "La contraseña no puede estar vacía.";
                        tError.Visibility = Visibility.Visible;
                    }
                    else if (tPass1.Password != tPass2.Password)
                    {
                        tError.Text = "Las contraseñas no coinciden.";
                        tError.Visibility = Visibility.Visible;
                    }
                    else
                    {
                        // Procesar cambio y mover al historial
                        XElement historico = usuario.Element("Historico") ?? new XElement("Historico");
                        if (historico.Parent == null) usuario.Add(historico);

                        var actuales = usuario.Elements("Password").ToList();
                        foreach (var p in actuales) { p.Remove(); p.SetAttributeValue("activa", null); historico.Add(p); }

                        usuario.Add(new XElement("Password", new XAttribute("activa", "true"), new XAttribute("fecha", DateTime.Now.ToString("s")), Encriptar(tPass1.Password)));

                        GuardarXML();
                        TxtPassword.Password = tPass1.Password;
                        validado = true;
                    }
                }
                else { validado = true; } // Usuario canceló
            }
        }

        private async void BtnHistorial_Click(object sender, RoutedEventArgs e)
        {
            var nodo = ArbolServidores.SelectedNode;
            if (nodo == null || !_nodosServidores.ContainsKey(nodo)) return;
            XElement? usuario = _nodosServidores[nodo].Descendants("Usuario").FirstOrDefault();
            if (usuario == null) return;

            var stack = new StackPanel { Spacing = 10, Width = 420, Padding = new Thickness(5) };
            var historico = usuario.Element("Historico")?.Elements("Password").OrderByDescending(x => x.Attribute("fecha")?.Value) ?? Enumerable.Empty<XElement>();

            foreach (var p in historico)
            {
                string pass = Desencriptar(p.Value);
                var grid = new Grid();
                grid.ColumnDefinitions.Add(new ColumnDefinition { Width = new GridLength(1, GridUnitType.Star) });
                grid.ColumnDefinitions.Add(new ColumnDefinition { Width = GridLength.Auto });

                var txt = new TextBox { Text = pass, Header = p.Attribute("fecha")?.Value, IsReadOnly = true, Width = 300 };
                var btn = new Button { Content = "Copiar", VerticalAlignment = VerticalAlignment.Bottom, Margin = new Thickness(5, 0, 0, 0) };
                btn.Click += (s, a) => { var dp = new DataPackage(); dp.SetText(pass); Clipboard.SetContent(dp); };

                grid.Children.Add(txt); Grid.SetColumn(btn, 1); grid.Children.Add(btn);
                stack.Children.Add(grid);
            }

            var dialog = new ContentDialog { Title = "Historial de Claves", Content = new ScrollViewer { Content = stack, MaxHeight = 400 }, CloseButtonText = "Cerrar", XamlRoot = this.Content.XamlRoot };
            await dialog.ShowAsync();
        }

        // --- Auxiliares UI y Cripto ---
        private void GridNodo_Loaded(object sender, RoutedEventArgs e)
        {
            if (sender is Grid g && g.DataContext is TreeViewNode n)
                if (g.FindName("BtnAddSrvRef") is Button b) b.Visibility = _nodosGrupos.ContainsKey(n) ? Visibility.Visible : Visibility.Collapsed;
        }

        private async void BtnAddGrupo_Click(object sender, RoutedEventArgs e)
        {
            var input = new TextBox { Header = "Nombre del Grupo" };
            var dialog = new ContentDialog { Title = "Nuevo Grupo", Content = input, PrimaryButtonText = "Crear", CloseButtonText = "Cancelar", XamlRoot = this.Content.XamlRoot };
            if (await dialog.ShowAsync() == ContentDialogResult.Primary && !string.IsNullOrWhiteSpace(input.Text))
            {
                _xmlData?.Root?.Add(new XElement("Grupo", new XAttribute("nombre", input.Text)));
                GuardarXML(); CargarDatosXML();
            }
        }

        private async void BtnInnerAddServidor_Click(object sender, RoutedEventArgs e)
        {
            var nodo = (sender as Button)?.Tag as TreeViewNode;
            if (nodo == null || !_nodosGrupos.ContainsKey(nodo)) return;
            // ... (Lógica de agregar servidor similar a la anterior, usando Encriptar())
            CargarDatosXML(); // Recargar al finalizar
        }

        private string Encriptar(string texto)
        {
            try
            {
                using Aes aes = Aes.Create();
                byte[] key = new byte[32]; Array.Copy(Encoding.UTF8.GetBytes(MasterKey), key, Math.Min(MasterKey.Length, 32));
                aes.Key = key; aes.IV = new byte[16];
                using var enc = aes.CreateEncryptor();
                byte[] buffer = Encoding.UTF8.GetBytes(texto);
                return Convert.ToBase64String(enc.TransformFinalBlock(buffer, 0, buffer.Length));
            }
            catch { return ""; }
        }

        private string Desencriptar(string base64)
        {
            try
            {
                using Aes aes = Aes.Create();
                byte[] key = new byte[32]; Array.Copy(Encoding.UTF8.GetBytes(MasterKey), key, Math.Min(MasterKey.Length, 32));
                aes.Key = key; aes.IV = new byte[16];
                using var dec = aes.CreateDecryptor();
                byte[] buffer = Convert.FromBase64String(base64.Trim());
                return Encoding.UTF8.GetString(dec.TransformFinalBlock(buffer, 0, buffer.Length));
            }
            catch { return "---"; }
        }

        private void CopiarPassword_Click(object sender, RoutedEventArgs e)
        {
            var dp = new DataPackage(); dp.SetText(TxtPassword.Password); Clipboard.SetContent(dp);
        }
